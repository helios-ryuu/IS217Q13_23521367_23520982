Bc 4: Training XGBoost Model

XGBoost Parameters:

  • objective='multi:softprob': Multi-class classification vi probability output
  • num_class=4: 4 classes (Severity 0-3, tng ng 1-4 ban u)
  • max_depth=6: su cy quyt nh (cao hn = phc tp hn, d overfit)
  • learning_rate=0.1: Tc hc (thp hn = hc chm nhng chnh xc hn)
  • n_estimators=100: S lng cy (nhiu hn = tt hn nhng chm hn)
  • tree_method='gpu_hist' hoc 'hist': S dng GPU nu c, khng th CPU

Thi gian d kin:

  • GPU (RTX 3050Ti): ~5-10 pht
  • CPU (i5-12500H): ~15-25 pht

import time

# Cu hnh XGBoost parameters
params = {
    'objective': 'multi:softprob',  # Multi-class classification
    'num_class': 4,                  # 4 classes (0-3)
    'max_depth': 6,                  #  su cy
    'learning_rate': 0.1,            # Learning rate (eta)
    'subsample': 0.8,                # Subsample 80% data cho mi tree
    'colsample_bytree': 0.8,         # Subsample 80% features cho mi tree
    'eval_metric': 'mlogloss',       # Multi-class logloss
    'random_state': 42,
    'n_jobs': -1                     # Dng tt c CPU cores
}

# Thm GPU support nu c
if USE_GPU:
    params['tree_method'] = 'gpu_hist'
    params['gpu_id'] = 0
    print(" Training vi GPU (RTX 3050Ti)...")
else:
    params['tree_method'] = 'hist'  # CPU-optimized
    print(" Training vi CPU (i5-12500H)...")

print(f"\n XGBoost Parameters:")
for key, value in params.items():
    print(f"  {key}: {value}")

# Train model
print(f"\n Bt u training lc: {time.strftime('%H:%M:%S')}")
start_time = time.time()

model = xgb.XGBClassifier(**params, n_estimators=100)

    X_train, y_train,
    eval_set=[(X_train, y_train), (X_test, y_test)],
    verbose=10  # In progress mi 10 iterations
)

# 5.3: ROC Curves (Receiver Operating Characteristic)
# ROC curve cho thy trade-off gia True Positive Rate v False Positive Rate
# AUC (Area Under Curve): Din tch di ng cong, cng gn 1.0 cng tt
# AUC = 0.5: Random guess, AUC = 1.0: Perfect classifier

from sklearn.preprocessing import label_binarize
from itertools import cycle

# Binarize labels cho multi-class ROC
y_test_bin = label_binarize(y_test, classes=[0, 1, 2, 3])
n_classes = 4

# Compute ROC curve v AUC cho tng class
fpr = dict()
tpr = dict()
roc_auc = dict()

for i in range(n_classes):
    fpr[i], tpr[i], _ = roc_curve(y_test_bin[:, i], y_pred_proba[:, i])
    roc_auc[i] = auc(fpr[i], tpr[i])

# Plot ROC curves
plt.figure(figsize=(12, 8))
colors = cycle(['blue', 'red', 'green', 'orange'])

for i, color in zip(range(n_classes), colors):
    plt.plot(fpr[i], tpr[i], color=color, lw=2,
             label=f'Severity {i+1} (AUC = {roc_auc[i]:.3f})')

plt.plot([0, 1], [0, 1], 'k--', lw=2, label='Random Guess (AUC = 0.5)')
